abstract class Transport{
	String TName;
	int arriveTime;
	int leaveTime;
	int getTTime;
	int securityTime;
	int waitTime;
	int Speed;
	double ticket;
	abstract public void Count(int distance);
	public void Display() {};
}

class Car extends Transport{
	int allTime;
	public Car(){
		TName= "Æû³µ";
		arriveTime=30;
		leaveTime=40;
		getTTime=10;
		securityTime=0;
		waitTime=10;
		Speed=80;
	}
	
	public void Count(int distance) {
		allTime=arriveTime+leaveTime+getTTime+securityTime+waitTime+distance/Speed;
	}
	
	public void Display() {
		System.out.print("Transportation:"+TName);
		System.out.println("The total time:"+allTime);
	}
}
